``` CSS (in styles.css file): ``` body { font-family: Arial, sans-serif; background-color: #f7f7f7; } header { background-color: #333; color: #fff; padding: 20px; text-align: center; } main { display: flex; flex-direction: column; align-items: center; padding: 20px; } .scenography { margin-bottom: 40px; } .stand, .decor { margin: 20px; padding: 20px; background-color: #fff; box-shadow: 0 0 10px rgba(0, 0, 0, 0.1); } .stand img, .decor img { width: 100%; height: 200px; object-fit: cover; } .stand p, .decor p { font-size: 16px; color: #666; } ``` JavaScript (in script.js file): ``` // Add a class to the body element to trigger the CSS animations document.body.classList.add('animate'); // Animate the exhibition stands and event decor elements const stands = document.querySelectorAll('.stand'); const decor = document.querySelectorAll('.decor'); stands.forEach((stand) => { stand.addEventListener('mouseover', () => { stand.classList.add('hover'); }); stand.addEventListener('mouseout', () => { stand.classList.remove('hover'); }); }); decor.forEach((decor) => { decor.addEventListener('mouseover', () => { decor.classList.add('hover'); }); decor.addEventListener('mouseout', () => { decor.classList.remove('hover'); }); }); ```
